 1#!/bin/sh
 2
 3# vstup
 4vis="LC81920252013215LGN00_B4"
 5nir="LC81920252013215LGN00_B5"
 6mapset="landsat"
 7# vysledek
 8ndvi="ndvi"
 9
10# pridat mapset do vyhledavaci cesty
11g.mapsets mapset=$mapset ope=add --q
12
13# nastavit vypocetni region
14g.region raster=$vis
15
16# vypocet NDVI
17echo "VIS: $vis ; NIR: $nir"
18r.mapcalc exp="$ndvi = float($nir - $vis) / ($nir + $vis)" --o
19
20# reklasifikace (1,2,3)
21echo "Reklasifikuji..."
22# r.reclass umi reklasifikovat pouze celociselne rastry, proto pouzime
23# r.recode
24r.recode input=$ndvi output=r_$ndvi rules=- --o <<EOF
25 -1:0.05:1 
26 0.05:0.35:2 
27 0.35:1:3
28EOF
29
30# popisky
31r.category map=r_$ndvi sep=':' rules=- <<EOF
321:bez vegetace, vodni plochy
332:plochy s minimalni vegetaci
343:plochy pokryte vegetaci
35EOF
36
37# tabulka barev
38r.colors map=r_$ndvi rules=- --q <<EOF
39 1 red
40 2 yellow
41 3 0 136 26
42EOF
43
44# vypsat vysledek
45echo "Generuji report..."
46r.stats -pl input=r_$ndvi
47
48echo "Hotovo!"
